## Prophet Usage Pattern

```python
from prophet import Prophet
import pandas as pd

def prophet_forecast(dates: list[str], values: list[float], horizon: int = 26) -> dict:
    """Forecast with Prophet. Returns P10/P50/P90 at key horizons."""
    df = pd.DataFrame({"ds": pd.to_datetime(dates), "y": values})

    model = Prophet(
        interval_width=0.80,      # 80% confidence interval
        yearly_seasonality=False,  # Weekly data — no annual seasonality
        weekly_seasonality=False,  # Engineering metrics lack weekly patterns
        changepoint_prior_scale=0.05,  # Conservative; prevents overfitting
    )
    model.fit(df)

    future = model.make_future_dataframe(periods=horizon, freq="W")
    forecast = model.predict(future)

    return {
        "p10": forecast["yhat_lower"].iloc[-horizon:].tolist(),
        "p50": forecast["yhat"].iloc[-horizon:].tolist(),
        "p90": forecast["yhat_upper"].iloc[-horizon:].tolist(),
    }
```

## Isolation Forest Pattern

```python
from sklearn.ensemble import IsolationForest
import numpy as np

def detect_anomalies(feature_matrix: np.ndarray, contamination: float = 0.1) -> list[bool]:
    """
    Returns True for anomalous weeks.
    feature_matrix: shape (n_weeks, n_features)
    contamination: expected fraction of anomalies (0.1 = 10%)
    """
    model = IsolationForest(contamination=contamination, random_state=42)
    labels = model.fit_predict(feature_matrix)
    return [label == -1 for label in labels]  # -1 = anomaly
```

## Change-Point Detection Pattern

```python
import ruptures as rpt

def detect_change_points(series: list[float], min_size: int = 4) -> list[int]:
    """
    Returns list of week indices where trend changed.
    min_size: minimum weeks between change-points.
    """
    signal = np.array(series)
    model = rpt.Pelt(model="rbf", min_size=min_size).fit(signal)
    breakpoints = model.predict(pen=1.0)  # pen: lower = more sensitive
    return breakpoints[:-1]  # Remove trailing sentinel
```

## Monte Carlo Pattern

```python
def monte_carlo_forecast(
    current_value: float,
    weekly_delta_mean: float,
    weekly_delta_std: float,
    horizon_weeks: int,
    n_simulations: int = 10_000,
) -> dict:
    """Returns P10/P50/P90 at horizon via Monte Carlo."""
    simulations = []
    for _ in range(n_simulations):
        path = current_value
        for _ in range(horizon_weeks):
            path += np.random.normal(weekly_delta_mean, weekly_delta_std)
        simulations.append(path)

    return {
        "p10": float(np.percentile(simulations, 10)),
```
